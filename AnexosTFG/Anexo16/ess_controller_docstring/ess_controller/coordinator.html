369    async def fetch_forecast_solar_data(self):
370        """Fetch solar forecast data from Forecast.Solar API."""
371        # Load data at startup
372        last_request_time, forecast_data = await self.async_load_data()
373
374        current_time = datetime.now()
375        try:
376            # Check if the minimum time between requests has passed
377            if last_request_time and \
378                (current_time - last_request_time <= FORECAST_UPDATE_INTERVAL):
379                self.logger.debug("Skipping request to Forecast.Solar to avoid frequent reads")
380                return forecast_data, last_request_time  # Avoid the call if the interval is less than the minimum allowed
381
382            assert self._session is not None, "HTTP session is not initialized"
383            url = await self.get_url_forecast_solar()
384
385            self.logger.debug(f"Starting request to {url}")
386        except Exception as e:
387            self.logger.error(f"Error comparing times with Forecast.Solar: {e}")
388            self.logger.error(f"Forecast.Solar Store last_request_time: {last_request_time}")
389            self.logger.error(f"Forecast.Solar Store forecast_data: {forecast_data}")
390            self.logger.error(f"fetch_forecast_solar_data current_time: {current_time}")
391            last_request_time = current_time
392            await self.async_save_data(current_time, forecast_data)
393            return forecast_data, last_request_time
394
395        try:
396            # Set a timeout for the request
397            async with async_timeout.timeout(10):
398                async with self._session.get(url) as resp:
399                    if resp.status == 200:
400                        data = await resp.json()
401                        # Update the last request time
402                        last_request_time = current_time
403                        # Log the estimated production data
404                        self.logger.debug("Newly downloaded ForeCastSolar data:")
405                        forecast_data = data.get("result", {}).get("watt_hours_period", None)
406                        # Save the data to persistent storage
407                        await self.async_save_data(last_request_time, forecast_data)
408
409                        return forecast_data, last_request_time  # Return the JSON data from the API
410                    elif resp.status == 429:
411                        self.logger.warning(f"Too many requests to Forecast.Solar, waiting {FORECAST_UPDATE_INTERVAL} h:m:s")
412                        last_request_time = current_time
413                        await self.async_save_data(current_time, forecast_data)
414                        return forecast_data, last_request_time
415                    else:
416                        self.logger.warning(f"Error in Forecast.Solar API: Status {resp.status}")
417                        return None, None
418
419        except aiohttp.ClientError as e:
420            self.logger.error(f"Connection error with Forecast.Solar: {e}")
421            return None, None
422        except asyncio.TimeoutError:
423            self.logger.error("Timeout while trying to connect to Forecast.Solar")
424            return None, None
425        except Exception as e:
426            self.logger.error(f"Unexpected error: {e}")
427            return None, None

370    async def fetch_forecast_solar_data(self):
371        """Fetch solar forecast data from Forecast.Solar API."""
372        # Load data at startup
373        last_request_time, forecast_data = await self.async_load_data()
374
375        current_time = datetime.now()
376        try:
377            # Check if the minimum time between requests has passed
378            if last_request_time and \
379                (current_time - last_request_time <= FORECAST_UPDATE_INTERVAL):
380                self.logger.debug("Skipping request to Forecast.Solar to avoid frequent reads")
381                return forecast_data, last_request_time  # Avoid the call if the interval is less than the minimum allowed
382
383            assert self._session is not None, "HTTP session is not initialized"
384            url = await self.get_url_forecast_solar()
385
386            self.logger.debug(f"Starting request to {url}")
387        except Exception as e:
388            self.logger.error(f"Error comparing times with Forecast.Solar: {e}")
389            self.logger.error(f"Forecast.Solar Store last_request_time: {last_request_time}")
390            self.logger.error(f"Forecast.Solar Store forecast_data: {forecast_data}")
391            self.logger.error(f"fetch_forecast_solar_data current_time: {current_time}")
392            last_request_time = current_time
393            await self.async_save_data(current_time, forecast_data)
394            return forecast_data, last_request_time
395
396        try:
397            # Set a timeout for the request
398            async with async_timeout.timeout(10):
399                async with self._session.get(url) as resp:
400                    if resp.status == 200:
401                        data = await resp.json()
402                        # Update the last request time
403                        last_request_time = current_time
404                        # Log the estimated production data
405                        self.logger.debug("Newly downloaded ForeCastSolar data:")
406                        forecast_data = data.get("result", {}).get("watt_hours_period", None)
407                        # Save the data to persistent storage
408                        await self.async_save_data(last_request_time, forecast_data)
409
410                        return forecast_data, last_request_time  # Return the JSON data from the API
411                    elif resp.status == 429:
412                        self.logger.warning(f"Too many requests to Forecast.Solar, waiting {FORECAST_UPDATE_INTERVAL} h:m:s")
413                        last_request_time = current_time
414                        await self.async_save_data(current_time, forecast_data)
415                        return forecast_data, last_request_time
416                    else:
417                        self.logger.warning(f"Error in Forecast.Solar API: Status {resp.status}")
418                        return None, None
419
420        except aiohttp.ClientError as e:
421            self.logger.error(f"Connection error with Forecast.Solar: {e}")
422            return None, None
423        except asyncio.TimeoutError:
424            self.logger.error("Timeout while trying to connect to Forecast.Solar")
425            return None, None
426        except Exception as e:
427            self.logger.error(f"Unexpected error: {e}")
428            return None, None
